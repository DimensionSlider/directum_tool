SELECT
vid.Name                'Имя справочника',	--0
vid.Kod                 'Код справочника',	--1
vidrecv.Name            'Заголовок реквизита',  --2
recvan.Name             'Имя реквизита',        --3
recvan.Kod              'Код реквизита',        --4
recvan.FldName          'Имя поля в бд',        --5
CASE recvan.Razd
        WHEN 'Ш'
        THEN 'Карточка'
        WHEN 'Т'
        THEN 'Таблица'
        WHEN 'С'
        THEN 'Таблица 2'
        WHEN 'Р'
        THEN 'Таблица 3'
        WHEN 'О'
        THEN 'Таблица 4'
        WHEN 'Н'
        THEN 'Таблица 5'
        WHEN 'М'
        THEN 'Таблица 6'
        WHEN 'К'
        THEN 'Действие'
end 'Раздел',                                   --6
'MBAnalit'              'Физ имя справочника',  --7

CASE recvan.Type
       WHEN 'Д'
       THEN 'Дата'
       WHEN 'Ч'
       THEN 'Дробное число'
       WHEN 'П'
       THEN 'Признак'
       WHEN 'А'
       THEN 'Справочник'
       WHEN 'С'
       THEN 'Строка'
       WHEN 'М'
       THEN 'Текст'
       WHEN 'Ц'
       THEN 'Целое число'
END                     'Тип',                   --8

case recvan.Stored
  WHEN 'Д'
  THEN '-1'
  WHEN 'Н'
  THEN '0'
end  'Сохранять', --9

recvan.PriznValues 'Значения признака', --10
case vidrecv.IsNull 
WHEN 'Д'
THEN '-1'
WHEN 'Н'
THEN '0'
end  'Обязательный' --11

FROM 
        MBVidAn vid
        INNER JOIN MBVidAnRecv vidrecv 
        INNER JOIN MBRecvAn recvan ON
        vidrecv.Kod = recvan.Kod ON 
        vid.Vid = vidrecv.Vid 

WHERE
      (  vid.Name like '%%' or vid.Kod like '%%' )
      and
      (  recvan.Name like '%%' or vidrecv.Name like '%%'  )

ORDER BY
        vid.Name ASC, recvan.Razd DESC, vidrecv.Name ASC
